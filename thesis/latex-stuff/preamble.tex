\mdfdefinestyle{bgstyle}{
	hidealllines=true,
  	skipabove=\topskip,  	
  	skipbelow=\topskip,
  	outermargin=15pt,
  	innermargin=15pt,
  	innerrightmargin=15pt,

\newtheoremstyle{myplain} %
	{3pt} % Space above
	{3pt} % Space below
	{\itshape} % Body font
	{} % Ident amount
	{\bfseries} % Theorem head font
	{\\} %  Punctuation after theorem head
	{0pt} % Space after theorem head [def: .5em]
	{} % Theorem head spec
\newtheoremstyle{mydefinition} %
	{3pt} % Space above
	{3pt} % Space below
	{\normalfont} % Body font
	{} % Ident amount
	{\bfseries} % Theorem head font
	{\\} %  Punctuation after theorem head
	{0pt} % Space after theorem head [def: .5em]
	{} % Theorem head spec
